    const firebaseConfig = {
      apiKey: "TU_API_KEY",                       // 游대 CAMBIA esto
      authDomain: "TU_PROYECTO.firebaseapp.com",  // 游대 CAMBIA esto
      databaseURL: "https://TU_PROYECTO.firebaseio.com", // 游대 CAMBIA esto

      storageBucket: "TU_PROYECTO.appspot.com",   // 游대 CAMBIA esto
      messagingSenderId: "XXXXXX",                // 游대 CAMBIA esto
      appId: "XXXXXX"                             // 游대 CAMBIA esto
    };

    firebase.initializeApp(firebaseConfig);
    const db = firebase.database();

    function actualizarEstado(sensor, activo) {
      const boton = document.getElementById(sensor);
      boton.className = "sensor-btn " + (activo ? "activo" : "inactivo");
    }

    function escucharSensores() {
      for (let i = 1; i <= 9; i++) {
        const sensorID = "sensor" + i;
        const ref = db.ref("sensores/" + sensorID);
        ref.on("value", (snapshot) => {
          const estado = snapshot.val();
          actualizarEstado(sensorID, estado == 1);
        });
      }
    }

    escucharSensores();
  </script>
